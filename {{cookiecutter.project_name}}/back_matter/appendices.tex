This appendix provides samples of list usage in LaTeX.

	\item Item entry
	\item Item entry
\end{itemize}

\subsection{Ordered Lists} \label{ordered-lists}

	\item Item 1
	\item Item 2
\end{enumerate}

\subsection{Alphabetical Lists} \label{alphabetical-list}

\end{enumerate}
