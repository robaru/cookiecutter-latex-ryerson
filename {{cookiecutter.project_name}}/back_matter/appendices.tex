%  === COMMANDS ===

% --- \appendices ---
\newcommand{\appendices}[1]{
	\setcounter{section}{0}
	\setcounter{figure}{0}
	\setcounter{table}{0}
	\refstepcounter{appendices}
	\par\chapter*{Appendix \theappendices. #1}
	\addcontentsline{loa}{section}
	{\protect\numberline{\theappendices}#1}\par
}
\renewcommand{\theappendices}{\Alph{appendices}}

% --- \subappendices ---
\newcommand{\subappendices}[1]{
	\refstepcounter{section}
	\par\section*{\theappendices\thesection. #1}
	\addcontentsline{loa}{subsection}
	{\protect\numberline{\theappendices\thesection}#1}\par
}

% --- Change figure, table, code numbering ---

\renewcommand\thefigure{\theappendices.\arabic{figure}}
\renewcommand\thetable{\theappendices.\arabic{table}}
\counterwithout{lstlisting}{section}
\renewcommand\thelstlisting{\theappendices.\arabic{lstlisting}}
