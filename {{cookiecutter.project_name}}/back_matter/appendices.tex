% === TEXT ===

\appendices{Text} \label{text}

This appendix provides samples of text usage in LaTeX.

\subappendices{Bold Text} \label{bold-text}

\textbf{Bold} paragraph text.

\subappendices{Italicized Text} \label{italicized-text}

\textit{Italicized} paragraph text.

\subappendices{Underlined Text} \label{underlined-text}

\underline{Underline} paragraph text.

% === CITATIONS ===

\appendices{Citations} \label{citations}

This appendix provides samples of citation usage in LaTeX.

\subappendices{Bracket Citations} \label{bracket-citations}

Citation \citep{lastnamep:yearp}. Multiple Citation \citep{lastnamep:yearp,lastnamee:yeare,lastnamea:yeara}.

\subappendices{Direct Citations} \label{direct-citations}

Citation \citet{lastnamep:yearp}. Multiple Citation \citet{lastnamep:yearp,lastnamee:yeare,lastnamea:yeara}.

% === FIGURES ===

\appendices{Figures} \label{figures}

This appendix provides a sample of graphic usage in LaTeX.

\begin{figure}[hb] \centering
	\includegraphics[width=3in]{graphic}
	\caption{\textbf{Figure Title.} Figure text.}
	\label{figure:1}
\end{figure}

Reference Figure \ref{figure:1}.

% === LISTS ===

\appendices{Lists} \label{lists}

This appendix provides samples of list usage in LaTeX.

\subappendices{Unordered Lists} \label{unordered-lists}

\begin{itemize}
	\item Item entry
	\item Item entry
\end{itemize}

\subappendices{Ordered Lists} \label{ordered-lists}

\begin{enumerate}
	\item Item 1
	\item Item 2
\end{enumerate}

\subappendices{Alphabetical Lists} \label{alphabetical-list}

\begin{enumerate}[label=(\alph*)]
	\item Item a
	\item Item b
\end{enumerate}
