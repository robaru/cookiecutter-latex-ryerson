% === \listofappendices ===
\newcommand{\listofappendicesname}{List of Appendices}
\newlistof{appendices}{loa}{\listofappendicesname}

% === \appendices ===
\newcommand{\appendices}[1]{
	\refstepcounter{appendices}
	\par\noindent\textbf{\theappendices. #1}
	\addcontentsline{loa}{section}
	{\protect\numberline{\theappendices.}#1}\par
}
\renewcommand{\theappendices}{\Alph{appendices}}

% === \subappendices ===
\newcommand{\subappendices}[1]{
	\refstepcounter{section}
	\par\noindent\textbf{\theappendices\thesection. #1}
	\addcontentsline{loa}{subsection}
	{\protect\numberline{\theappendices\thesection.}#1}\par
}
